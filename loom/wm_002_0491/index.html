20. Server

Will instantiate a local http server on port 8080 by default. Only enabled if WWSERVER is defined.

<<static_funcdefs>>=
int weewiki_server(weewiki_d *ww, int argc, char *argv[]);
<<argparse_entries>>=
{"server", 6, p_server, NULL},
<<static_funcdefs>>=
#ifdef WWSERVER
static int p_server(int argc, char *argv[]);
#endif
<<functions>>=
#ifdef WWSERVER
static int p_server(int argc, char *argv[])
{
    weewiki_d ww;
    return weewiki_server(&ww, argc, argv);
}
#endif
