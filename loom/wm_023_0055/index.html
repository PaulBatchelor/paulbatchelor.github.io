7. Operations

7.1. Num

Creates a 7-bit number in a single instruction.

<<funcdefs>>=
int bitlang_num(bitlang_state *st, int num);
<<funcs>>=
int bitlang_num(bitlang_state *st, int num)
{
    num &= 0x7f;

    if (st->len >= st->sz) return 1;

    st->bytes[st->len] = 0x80 | num;
    st->len++;
    return 0;
}
