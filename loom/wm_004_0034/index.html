4.3. Startup

<<funcdefs>>=

<<functions>>=
int wwserver_init(wwserver_d *wws, int port)
{
    wws->server = http_server_init(port, handle_request);
    wws->env = weewiki_janet_setup();
    wws->use_sqlar = 0;
    return 1;
}
