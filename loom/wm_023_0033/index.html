<<funcs>>=
void bitlang_state_init(bitlang_state *st, char *b, int sz)
{
    int i;
    st->bytes = b;
    st->sz = sz;
    st->len = 0;

    for (i = 0; i < sz; i++) {
        st->bytes[i] = 0;
    }
}
