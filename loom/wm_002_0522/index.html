23. Zet

Zet is the weewiki zettelkasten. This is the entry point for the zet command-line interface.

<<argparse_entries>>=
{"zet", 3, p_zet, NULL},
<<static_funcdefs>>=
static int p_zet(int argc, char *argv[]);
<<functions>>=
int ww_zet(int argc, char *argv[]);
static int p_zet(int argc, char *argv[])
{
    return ww_zet(argc, argv);
}
