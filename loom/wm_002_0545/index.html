26. Clear

Clears a weewiki database.

<<argparse_entries>>=
{"clear", 5, p_clear, NULL},
<<static_funcdefs>>=
static int p_clear(int argc, char *argv[]);
<<functions>>=
static int p_clear(int argc, char *argv[])
{
    weewiki_d ww;
    weewiki_init(&ww);
    weewiki_open(&ww, weewiki_dbname_get());

    weewiki_clear(&ww);

    weewiki_close(&ww);
    weewiki_clean(&ww);
    return 0;
}
