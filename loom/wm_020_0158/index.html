void seqvm_reset(seqvm *svm)
{
    int i;

    if (svm->blkpos >= 0) {
        svm->blkpos = -1;
        svm->main = svm->next;
    }

    for (i = 0; i < 4; i++) {
        svm->main.chan[i].tick = 0;
        svm->next.chan[i].tick = 0;
    }
}
