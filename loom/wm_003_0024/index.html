4. Parsing Org code in Janet

<<static_funcdefs>>=
static Janet cfun_org(int32_t argc, Janet *argv);
<<functions>>=
static Janet cfun_org(int32_t argc, Janet *argv)
{
    int32_t len;
    weewiki_d *ww;
    const char *txt;
    ww = weewiki_get();
    txt = (const char *)janet_to_string(argv[0]);
    len = strlen(txt);
    weewiki_janet_org(ww, txt, len);
    return janet_wrap_nil();
}
