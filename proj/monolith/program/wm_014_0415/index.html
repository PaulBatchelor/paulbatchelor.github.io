static runt_int rproc_trigclk(runt_vm *vm, runt_ptr p)
{
    monolith_d *m;
    runt_int rc;
    rpw_param in;
    pw_patch *patch;
    pw_node *node;
    const char *name;
    runt_stacklet *s;
    monolith_page *pg;
    page_trig_d *trig;

    m = runt_to_cptr(p);
    rc = runt_ppop(vm, &s);
    RUNT_ERROR_CHECK(rc);

    name = runt_to_string(s->p);

    rc = rpw_get_param(vm, &in);
    RUNT_ERROR_CHECK(rc);

    rc = runt_monolith_lookup_page(vm, m,
                                   name, "trig",
                                   is_trig, &pg);
    RUNT_ERROR_CHECK(rc);

    trig = monolith_page_data_get(pg);

    patch = monolith_patchwerk_get(m);
    rc = pw_patch_new_node(patch, &node);
    PW_RUNT_ERROR_CHECK(rc);
    node_trigclk(node, trig);

    rpw_set_param(vm, node, &in, 0);
    return RUNT_OK;
}

10.4. Trigwget Word

<<trig_runt_keywords>>=
monolith_runt_keyword(m, "trigwget", 8, rproc_trigwget, m);
<<trig_runt_functions>>=
static runt_int rproc_trigwget(runt_vm *vm, runt_ptr p)
{
    monolith_d *m;
    runt_int rc;
    pw_patch *patch;
    pw_node *node;
    const char *name;
    runt_stacklet *s;
    monolith_page *pg;
    page_trig_d *trig;
    rpw_param pwire;
    int wire;
    runt_stacklet *out;

    m = runt_to_cptr(p);
    rc = runt_ppop(vm, &s);
    RUNT_ERROR_CHECK(rc);

    name = runt_to_string(s->p);

    rc = runt_monolith_lookup_page(vm, m,
                                   name, "trig",
                                   is_trig, &pg);
    RUNT_ERROR_CHECK(rc);

    rc = rpw_get_param(vm, &pwire);
    RUNT_ERROR_CHECK(rc);

    if (!rpw_param_is_constant(&pwire)) {
        runt_print(vm, "wire must be a constant\n");
        return RUNT_NOT_OK;
    }

    wire = rpw_param_get_constant(&pwire);

    rc = runt_ppush(vm, &out);
    RUNT_ERROR_CHECK(rc);

    trig = monolith_page_data_get(pg);

    patch = monolith_patchwerk_get(m);
    rc = pw_patch_new_node(patch, &node);
    PW_RUNT_ERROR_CHECK(rc);
    rc = node_trigwget(node, trig, wire);
    PW_RUNT_ERROR_CHECK(rc);

    rpw_push_output(vm, node, out, 0);
    return RUNT_OK;
}
