2. Tangled Files

2.1. core

ftlist.c and ftlist.h are the core files.

<<ftlist.c>>=
#include <stdlib.h>
#include <math.h>
#include "soundpipe.h"
#define FTLIST_PRIV
#include "ftlist.h"
<<funcs>>
<<ftlist.h>>=
#ifndef FTLIST_H
#define FTLIST_H
<<typedefs>>
<<funcdefs>>
#ifdef FTLIST_PRIV
<<structs>>
#endif
#endif

This assumes that soundpipe.h has been included already.

2.2. runt

runt_ftlist.c and runt_ftlist.h creates runt bindings and words. Defining FTLIST_PRIV exposes the struct contents of sp_ftlist.

<<runt_ftlist.c>>=
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include "soundpipe.h"
#include "ftlist.h"
#include "runt.h"
#include "patchwerk.h"
#include "runt_patchwerk.h"
#include "runt_ftlist.h"
#include "runt_ftbl.h"
<<runt_macros>>
<<runt_funcs>>
<<runt_ftlist.h>>=
#ifndef RUNT_FTLIST_H
#define RUNT_FTLIST_H
<<runt_funcdefs>>
#endif

2.3. tchoose

tchoose.c implements a sample node that modifies the ftlist. A runt word loader is also provided.

<<tchoose.c>>=
#include <math.h>
#include "soundpipe.h"
#include "runt.h"
#include "patchwerk.h"
#include "runt_patchwerk.h"
#include "ftlist.h"
#include "runt_ftlist.h"
<<tchoose_funcdefs>>
<<tchoose_static_funcdefs>>
<<tchoose_funcs>>
