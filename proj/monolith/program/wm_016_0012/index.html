2. CLI function

The CLI can be instantiated with run_cfloop. It expects arguments from main.

<<main>>=
int run_cfloop(int argc, char *argv[])
{
    if(argc < 4) {
        printf("%s: in.wav out.wav cf\n", argv[0]);
        return 1;
    }

    return do_cfloop(argv[1], argv[2], atof(argv[3]));
}
