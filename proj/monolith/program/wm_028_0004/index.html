<<fbm.c>>=
#include <math.h>
#include <stdio.h>
#include "runt.h"
#include "patchwerk.h"
#include "monolith.h"
<<structs>>
<<funcdefs>>
<<static_funcdefs>>
<<funcs>>

#include <stdio.h>
