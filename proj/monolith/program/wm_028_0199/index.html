These are core words and functions designed to integrate ftlists into runt and patchwerk.

The loader for all the runt words is done with load_ftlist. Like all runt patchwerk loaders, this expects the patchwerk runt pointer pw.

7.2. ftlist_new

The runt word ftlist_new allocates a new ftlist using patchwerk, then pushes it onto the stack.

<<words>>=
runt_keyword_define(vm,
                    "ftlist_new",
                    10,
                    rproc_ftlist_new,
                    &c);
runt_cell_data(vm, c, pw);
<<wordfuncs>>=
static runt_int rproc_ftlist_new(runt_vm *vm, runt_ptr p)
{
    pw_patch *patch;
    sp_ftlist *ftlst;
    int rc;
    runt_stacklet *s;

    patch = rpw_get_patch(p);
    rc = runt_ppush(vm, &s);
    RUNT_ERROR_CHECK(rc);

    ftlst = malloc(sp_ftlist_sizeof());
    sp_ftlist_init(ftlst);

    rpw_append_ftlist(patch, ftlst);
    rpw_stacklet_ftlist(vm, s, ftlst);
    return RUNT_OK;
}

7.3. rpw_get_ftlist

Pops an ftlist from the runt stack, and does some light type checking.

<<runt_funcdefs>>=
runt_int rpw_get_ftlist(runt_vm *vm, sp_ftlist **ftl);
<<runt_funcs>>=
runt_int rpw_get_ftlist(runt_vm *vm, sp_ftlist **ftl)
{
    runt_int rc;
    runt_stacklet *s;

    rc = runt_ppop(vm, &s);
    RUNT_ERROR_CHECK(rc);
    if(s->t != FTLIST_TYPE) {
        runt_print(vm, "Parameter does not seem to be an ftlist\n");
        *ftl = NULL;
        return RUNT_NOT_OK;
    }
    *ftl = runt_to_cptr(s->p);
    return RUNT_OK;
}

7.4. rpw_append_ftlist

Allocates and appends an ftlist to the current patchwerk pointer list.

<<runt_funcdefs>>=
void rpw_append_ftlist(pw_patch *patch, sp_ftlist *ftl);
<<runt_funcs>>=
static void free_ftlist(pw_pointer *p)
{
    sp_ftlist *ftlst;

    ftlst = pw_pointer_data(p);
    sp_ftlist_clean(ftlst);
    free(ftlst);
}
void rpw_append_ftlist(pw_patch *patch, sp_ftlist *ftl)
{
    pw_patch_append_userdata(patch, free_ftlist, ftl);
}

7.5. type id

The type id is used inside the stacklet as a means for soft type checking. It it send to be END + 111, where END is the the last id of the default ugens (retrived via rpw_get_end).

This type ID represented as a macro FTLIST_TYPE.

<<runt_macros>>=
#define FTLIST_TYPE (rpw_get_end() + 111)

7.6. rpw_stacklet_ftlist

Configures a runt stacklet (presumably being pushed) to encapsulate an ftlist. The most important thing here is the type id, which is set to be FTLIST_TYPE.

<<runt_funcdefs>>=
void rpw_stacklet_ftlist(runt_vm *vm,
                         runt_stacklet *s,
                         sp_ftlist *ftl);
<<runt_funcs>>=
void rpw_stacklet_ftlist(runt_vm *vm,
                         runt_stacklet *s,
                         sp_ftlist *ftl)
{
    runt_stacklet_init(vm, s);
    s->p = runt_mk_cptr(vm, ftl);
    s->t = FTLIST_TYPE;
}
