Used to assign the griffin to any knob from any lane.

<<page_knobs_contents>>=

<<knobs_function_declarations>>=
static void knobs_aux_knob(page_knobs_d *knobs,
                           int lane,
                           int x, int y);
<<knobs_functions>>=
static void knobs_aux_knob(page_knobs_d *knobs,
                           int lane,
                           int x, int y)
{
    int k;

    k = (lane * 4 + x) + y * 16;

    if (k < 0) return;
    if (k >= 128) return;

    knobs->aux_knob = k;
}
