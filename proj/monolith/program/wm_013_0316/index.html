static runt_int rproc_seq16clk(runt_vm *vm, runt_ptr p)
{
    monolith_d *m;
    runt_int rc;
    rpw_param clk;
    pw_patch *patch;
    pw_node *node;
    const char *name;
    runt_stacklet *s;
    monolith_page *pg;
    page_seq16_d *seq16;

    m = runt_to_cptr(p);
    rc = runt_ppop(vm, &s);
    RUNT_ERROR_CHECK(rc);

    name = runt_to_string(s->p);

    rc = rpw_get_param(vm, &clk);
    RUNT_ERROR_CHECK(rc);

    rc = runt_monolith_lookup_page(vm, m, name, "seq16", is_seq16, &pg);
    RUNT_ERROR_CHECK(rc);

    seq16 = monolith_page_data_get(pg);

    patch = monolith_patchwerk_get(m);
    rc = pw_patch_new_node(patch, &node);
    PW_RUNT_ERROR_CHECK(rc);
    node_seq16clk(node, seq16);
    rpw_set_param(vm, node, &clk, 0);
    return RUNT_OK;
}
