4. eval

4.1. In C

<<function_declarations>>=
void monolith_janet_eval(monolith_d *m, const char *str);
<<functions>>=
void monolith_janet_eval(monolith_d *m, const char *str)
{
    JanetTable *env;
    env = monolith_janet_env(m);
    if (env == NULL) {
        fprintf(stderr, "Janet is not initialized yet.\n");
        return;
    }
    janet_dostring(env, str, NULL, NULL);
}
