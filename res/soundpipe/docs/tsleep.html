#include <stdlib.h>
#include <stdio.h>
#include <time.h>
#include "soundpipe.h"

typedef struct {
    sp_tsleep *tsleep;
    sp_osc *osc;
    sp_ftbl *ft; 
    int gate;
} UserData;

void process(sp_data *sp, void *udata) {
    UserData *ud = udata;
    SPFLOAT osc = 0, tsleep = 0;
    SPFLOAT tick = (sp->pos == 0) ? 1 : 0;
    sp_osc_compute(sp, ud->osc, NULL, &osc);
    sp_tsleep_compute(sp, ud->tsleep, &tick, &tsleep);
    if(tsleep) ud->gate = 1;
    sp->out[0] = osc * ud->gate;
}

int main() {
    srand(1234567);
    UserData ud;
    ud.gate = 0;
    sp_data *sp;
    sp_create(&sp);

    sp_tsleep_create(&ud.tsleep);
    sp_osc_create(&ud.osc);
    sp_ftbl_create(sp, &ud.ft, 2048);

    sp_tsleep_init(sp, ud.tsleep);
    ud.tsleep->time = 2.0;
    sp_gen_sine(sp, ud.ft);
    sp_osc_init(sp, ud.osc, ud.ft, 0);

    sp->len = 44100 * 5;
    sp_process(sp, &ud, process);

    sp_tsleep_destroy(&ud.tsleep);
    sp_ftbl_destroy(&ud.ft);
    sp_osc_destroy(&ud.osc);

    sp_destroy(&sp);
    return 0;
}
