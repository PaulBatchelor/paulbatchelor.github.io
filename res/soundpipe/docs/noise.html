noise
Files: noise.h, noise.c

White noise generator.

Functions

sp_noise_create(sp_noise **noise)
sp_noise_init(sp_data *sp, sp_noise *noise)
sp_noise_compute(sp_data *sp, sp_noise *noise, SPFLOAT *dummy, SPFLOAT *out)
sp_noise_destroy(sp_noise **noise)

Optional Parameters

gain: Amplitude. (Value between 0-1).
(Default value: 1)

Inputs

dummy: This doesn't do anything

Outputs

out: Signal out.

Example Code

#include <stdlib.h>
#include <stdio.h>
#include <time.h>
#include "soundpipe.h"

typedef struct {
    sp_noise *ns;
} udata;

void write_noise(sp_data *data, void *ud) {
    udata *udata = ud;
    SPFLOAT in = 0;
    sp_noise_compute(data, udata->ns, NULL, &data->out[0]);
}
int main() {
    srand(time(NULL));
    udata ud;
    sp_data *sp;
    sp_create(&sp);
    sp_noise_create(&ud.ns);
    sp_noise_init(sp, ud.ns);
    sp->len = 44100 * 5;
    sp_process(sp, &ud, write_noise);
    sp_noise_destroy(&ud.ns);
    sp_destroy(&sp);
    return 0;
}
