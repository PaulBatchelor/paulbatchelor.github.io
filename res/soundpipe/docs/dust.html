dust
Files: dust.h, dust.c

Dust creates a series of random impulses.

Functions

sp_dust_create(sp_dust **dust)
sp_dust_init(sp_data *sp, sp_dust *dust, SPFLOAT amp, SPFLOAT density)
sp_dust_compute(sp_data *sp, sp_dust *dust, SPFLOAT *dummy, SPFLOAT *out)
sp_dust_destroy(sp_dust **dust)

Mandatory Parameters

amp:
(Recommended value: 0.3)
density:
(Recommended value: 10)

Optional Parameters

bipolar: Bipolar flag. A non-zero makes the signal bipolar as opposed to unipolar.
(Default value: 0)

Inputs

dummy: This doesn't do anything.

Outputs

out: Signal out.

Example Code

#include <stdlib.h>
#include <stdio.h>
#include <time.h>
#include "soundpipe.h"

typedef struct {
    sp_dust *dst;
} udata;

void write_dust(sp_data *data, void *ud) {
    udata *udata = ud;
    SPFLOAT in = 0;
    sp_dust_compute(data, udata->dst, NULL, &data->out[0]);
}
int main() {
    srand(time(NULL));
    udata ud;
    sp_data *sp;
    sp_create(&sp);
    sp_dust_create(&ud.dst);
    sp_dust_init(sp, ud.dst, 0.3, 10);
    sp->len = 44100 * 5;
    ud.dst->bipolar = 1.0;
    sp_process(sp, &ud, write_dust);
    sp_dust_destroy(&ud.dst);
    sp_destroy(&sp);
    return 0;
}
