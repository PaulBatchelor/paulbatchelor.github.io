delay1
Files: delay1.h, delay1.c

Delays signal by a single sample. There's not much more to say on this. Mostly useful for filters and hacks.

Functions

Outputs

#include <stdlib.h>
#include <stdio.h>
#include <time.h>
#include "soundpipe.h"

typedef struct {
    sp_delay1 *delay1;
    sp_osc *osc;
    sp_ftbl *ft; 
} UserData;

void process(sp_data *sp, void *udata) {
    UserData *ud = udata;
    SPFLOAT osc = 0, delay1 = 0;
    sp_osc_compute(sp, ud->osc, NULL, &osc);
    sp_delay1_compute(sp, ud->delay1, &osc, &delay1);
    sp->out[0] = delay1;
}

int main() {
    srand(1234567);
    UserData ud;
    sp_data *sp;
    sp_create(&sp);

    sp_delay1_create(&ud.delay1);
    sp_osc_create(&ud.osc);
    sp_ftbl_create(sp, &ud.ft, 2048);

    sp_delay1_init(sp, ud.delay1);
    sp_gen_sine(sp, ud.ft);
    sp_osc_init(sp, ud.osc, ud.ft, 0);

    sp->len = 44100 * 5;
    sp_process(sp, &ud, process);

    sp_delay1_destroy(&ud.delay1);
    sp_ftbl_destroy(&ud.ft);
    sp_osc_destroy(&ud.osc);

    sp_destroy(&sp);
    return 0;
}
